IfBumped				//Display a message (Based on content number)
  SetTargetToWhoeverBumped
  IfTargetIsAPlayer
    IfStateIs0				//Not looping
      GetContent
      tmpargument = tmpargument
      SendMessageNear
      tmpargument = 100
      SetTime
      tmpargument = 1
      SetState

IfTimeOut				//Disable looping of messages
  tmpargument = 0
  SetState

End					// All done
