// This drops several enemies when the passage is blocked
// The water must have been filled at least once for it to function
IfTimeOut
  tmpargument = 20
  SetTime

  IfStateIs0
    // Check if the water has been filled
    GetWaterLevel
    tmpx = tmpargument
    tmpy = 600
    IfXIsMoreThanY
      // The water has been filled
      tmpargument = 1
      SetState
  Else
    tmpargument = passage
    SetTargetToWhoeverIsInPassage
      IfTargetIsAlive
        tmpargument = 0
        SendMessageNear

        // The drop sound
        tmpargument = 0
        PlaySound

        // Spawn the characters
        GetContent                 // The specific model slot

        // Character 1
        tmpx = selfx
        tmpy = selfy
        tmpdistance = rand & 255 + 800
        tmpturn = rand
        SpawnExactCharacterXYZ

        // Character 2
        tmpx = selfx - 256
        tmpdistance = rand & 255 + 800
        tmpturn = rand
        SpawnExactCharacterXYZ

        // Character 3
        tmpx = selfx + 256
        tmpdistance = rand & 255 + 800
        tmpturn = rand
        SpawnExactCharacterXYZ

        // Don't hang around
        GoPoof

End					// All done
