IfGrabbed
  SetTargetToWhoeverIsHolding		  //
  IfTargetIsAPlayer			  //
    tmpargument = 0			  //
    SendMessageNear			  //

IfUsed
  IfStateIs0
    SetTargetToWhoeverIsHolding
    tmpargument = [UNDE]
    IfTargetHasID
      tmpargument = 2           //special message for zombor
    Else
      tmpargument = 1
    SetState
    SendMessageNear
    tmpargument = 100
    SetTime

IfTimeOut
  tmpargument = 100
  SetTime
  tmpargument = 0
  SetState

IfHitGround				// Make a sound
  tmpargument = 0			  //
  PlaySound				  //

End					// All done
